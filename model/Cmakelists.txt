cmake_minimum_required(VERSION 3.18)
project(MiniLLMTorch LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Torch REQUIRED)

add_executable(inference_torch inference_torch.cpp)
target_link_libraries(inference_torch PRIVATE "${TORCH_LIBRARIES}")
target_compile_options(inference_torch PRIVATE ${TORCH_CXX_FLAGS})

add_executable(inference_libtorch inference.cpp)
target_link_libraries(inference_libtorch PRIVATE "${TORCH_LIBRARIES}")
target_compile_options(inference_libtorch PRIVATE ${TORCH_CXX_FLAGS})

if (MSVC)
  file(GLOB TORCH_DLLS "${TORCH_INSTALL_PREFIX}/lib/*.dll")
  foreach(target inference_torch inference_libtorch)
    add_custom_command(TARGET ${target}
      POST_BUILD
      COMMAND ${CMAKE_COMMAND} -E copy_if_different
      ${TORCH_DLLS}
      $<TARGET_FILE_DIR:${target}>)
  endforeach()
endif()
